Hastane Randevu ve Tahlil Sistemi için bir PSEUDOCODE yazmama yardım eder misin? İki ayrı modülü ayrı ayrı çalışması gerek. 
ilk önce hastane randevusu alma pseudocodesini daha sonra tahlil sisteminin pseudocodenu yazmamız lazım. son olarak. bu iki modülü birleştir.
BEGIN RandevuAl
    DISPLAY "Hoş geldiniz. Lütfen bilgilerinizi giriniz."
    INPUT hastaAd, hastaSoyad, TCKimlikNo
    DISPLAY "Bölüm Seçiniz: Kardiyoloji, Dahiliye, Ortopedi, Göz, Kulak Burun Boğaz"
    INPUT bolum
    DISPLAY "Tarih seçiniz (gg/aa/yyyy):"
    INPUT randevuTarihi
    DISPLAY "Saat seçiniz (09:00 - 17:00 arası):"
    INPUT randevuSaati

    IF randevuMüsaitMi(randevuTarihi, randevuSaati, bolum) == TRUE THEN
        randevuID ← RandevuKaydet(hastaAd, hastaSoyad, TCKimlikNo, bolum, randevuTarihi, randevuSaati)
        DISPLAY "Randevunuz başarıyla oluşturuldu. Randevu No: " + randevuID
    ELSE
        DISPLAY "Seçtiğiniz saat dolu. Lütfen başka bir saat seçiniz."
    END IF
END
BEGIN TahlilSistemi
    DISPLAY "Tahlil Sistemi'ne Hoş Geldiniz"
    INPUT TCKimlikNo

    IF HastaVarMi(TCKimlikNo) == TRUE THEN
        DISPLAY "Yapılacak tahlili seçiniz: Kan, İdrar, MR, Röntgen"
        INPUT tahlilTuru
        DISPLAY "Tahlil tarihi giriniz (gg/aa/yyyy):"
        INPUT tahlilTarihi

        tahlilID ← TahlilKaydet(TCKimlikNo, tahlilTuru, tahlilTarihi)
        DISPLAY "Tahlil kaydı oluşturuldu. Tahlil ID: " + tahlilID
    ELSE
        DISPLAY "Hasta kaydı bulunamadı. Lütfen önce randevu alınız."
    END IF
END
BEGIN HastaneSistemi
    DISPLAY "1 - Randevu Al"
    DISPLAY "2 - Tahlil Yaptır"
    INPUT secim

    IF secim == 1 THEN
        CALL RandevuAl
    ELSE IF secim == 2 THEN
        CALL TahlilSistemi
    ELSE
        DISPLAY "Geçersiz seçim."
    END IF
END

bu pseudocode yi dot diline çevir.
digraph HastaneRandevuVeTahlilSistemi {
    rankdir=TB;
    node [shape=box, style=rounded];

    // Giriş ve Menü
    Start [label="Sistem Başlat"];
    Menu [label="Kullanıcıya Seçim Sun:\n1 - Randevu Al\n2 - Tahlil Yaptır"];
    Secim1 [label="Seçim: 1 - Randevu Al"];
    Secim2 [label="Seçim: 2 - Tahlil Yaptır"];
    Gecersiz [label="Geçersiz Seçim"];

    // Randevu Alma Modülü
    R1 [label="Bilgileri Al:\nAd, Soyad, TC"];
    R2 [label="Bölüm Seçimi"];
    R3 [label="Tarih ve Saat Seçimi"];
    R4 [label="Randevu Müsait Mi?"];
    R5 [label="Randevu Oluştur\nve ID Göster"];
    R6 [label="Saat Dolu\nUyarı Göster"];

    // Tahlil Modülü
    T1 [label="TC Kimlik No Gir"];
    T2 [label="Hasta Var mı?"];
    T3 [label="Tahlil Türü Seç"];
    T4 [label="Tahlil Tarihi Gir"];
    T5 [label="Tahlil Kaydı Oluştur\nve ID Göster"];
    T6 [label="Hasta Yok\nRandevu Alın Uyarısı"];

    End [label="Sistem Sonu"];

    // Ana Menüden Seçim
    Start -> Menu;
    Menu -> Secim1 [label="seçim = 1"];
    Menu -> Secim2 [label="seçim = 2"];
    Menu -> Gecersiz [label="diğer"];

    // Randevu Akışı
    Secim1 -> R1;
    R1 -> R2;
    R2 -> R3;
    R3 -> R4;
    R4 -> R5 [label="evet"];
    R4 -> R6 [label="hayır"];
    R5 -> End;
    R6 -> End;

    // Tahlil Akışı
    Secim2 -> T1;
    T1 -> T2;
    T2 -> T3 [label="evet"];
    T2 -> T6 [label="hayır"];
    T3 -> T4;
    T4 -> T5;
    T5 -> End;
    T6 -> End;

    Gecersiz -> End;
}
